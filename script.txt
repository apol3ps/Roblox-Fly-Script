-- ============ ЛОГИКА ПОЛЁТА ============
uis.InputBegan:Connect(function(i, g)
    if g then return end
    local k = i.KeyCode
    if k == Enum.KeyCode.W then flyKeys.f = 1
    elseif k == Enum.KeyCode.S then flyKeys.b = 1
    elseif k == Enum.KeyCode.A then flyKeys.l = 1
    elseif k == Enum.KeyCode.D then flyKeys.r = 1
    end
end)

uis.InputEnded:Connect(function(i)
    local k = i.KeyCode
    if k == Enum.KeyCode.W then flyKeys.f = 0
    elseif k == Enum.KeyCode.S then flyKeys.b = 0
    elseif k == Enum.KeyCode.A then flyKeys.l = 0
    elseif k == Enum.KeyCode.D then flyKeys.r = 0
    end
end)

local function setFly(on)
    flyState = on
    local c = p.Character
    if not c or not c.Parent then return end
    local torso = c:FindFirstChild("UpperTorso") or c:FindFirstChild("Torso")
    local h = c:FindFirstChild("Humanoid")
    if not torso or not h then return end
    
    if flyState then
        vel = Instance.new("BodyVelocity")
        vel.Parent = torso
        vel.maxForce = Vector3.new(9e9, 9e9, 9e9)
        vel.Velocity = Vector3.new(0, 0.1, 0)
        
        gyro = Instance.new("BodyGyro")
        gyro.Parent = torso
        gyro.maxTorque = Vector3.new(9e9, 9e9, 9e9)
        gyro.P = 9e4
        gyro.CFrame = torso.CFrame
        
        h.PlatformStand = true
        activeFlyBtn.Text = "🔴 ПОЛЁТ АКТИВЕН"
        activeFlyBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 100)
    else
        if vel then vel:Destroy() end
        if gyro then gyro:Destroy() end
        if h then h.PlatformStand = false end
        activeFlyBtn.Text = "🛸 АКТИВИРОВАТЬ ПОЛЁТ"
        activeFlyBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 150)
    end
end

-- ЦИКЛ ПОЛЁТА
rs.Heartbeat:Connect(function()
    if not flyState then return end
    local c = p.Character
    if not c then return end
    local torso = c:FindFirstChild("UpperTorso") or c:FindFirstChild("Torso")
    local h = c:FindFirstChild("Humanoid")
    if not torso or not h then return end
    
    local cam = workspace.CurrentCamera
    local move = Vector3.new(flyKeys.r - flyKeys.l, 0, flyKeys.f - flyKeys.b)
    if move.Magnitude > 0 then move = move.Unit end
    
    local velVec = (cam.CFrame.RightVector * move.X + cam.CFrame.LookVector * move.Z) * 50 * flySpeed
    if vel then vel.Velocity = velVec end
    if gyro then gyro.CFrame = cam.CFrame end
end)
