    local function setWalkSpeed(value)
        walkSpeed = math.clamp(value, 0, 250)
        walkValue.Text = tostring(walkSpeed)
        local c = p.Character
        if c and c:FindFirstChild("Humanoid") then
            c.Humanoid.WalkSpeed = walkSpeed
        end
    end

    local function setJumpPower(value)
        jumpPower = math.clamp(value, 0, 500)
        jumpValue.Text = tostring(jumpPower)
        local c = p.Character
        if c and c:FindFirstChild("Humanoid") then
            c.Humanoid.JumpPower = jumpPower
        end
    end
