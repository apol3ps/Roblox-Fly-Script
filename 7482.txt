    function switchTab(tab)
        flyCont.Visible = (tab == "fly")
        speedCont.Visible = (tab == "speed")
        playerCont.Visible = (tab == "player")
        visualCont.Visible = (tab == "visual")
        espCont.Visible = (tab == "esp")
        infoCont.Visible = (tab == "info")
        local btns = {flyBtn, speedBtn, playerBtn, visualBtn, espBtn, infoBtn}
        local names = {"fly", "speed", "player", "visual", "esp", "info"}
        for i, btn in ipairs(btns) do
            if names[i] == tab then
                btn.TextColor3 = Color3.fromRGB(255, 100, 200)
                btn.BackgroundTransparency = 0
            else
                btn.TextColor3 = Color3.fromRGB(180, 180, 200)
                btn.BackgroundTransparency = 0.5
            end
        end
    end
