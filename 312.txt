        if p.Character and p.Character:FindFirstChild("HumanoidRootPart") and plr.Character and plr.Character:FindFirstChild("Head") and plr.Character.Head:FindFirstChild("NameTag") and plr.Character:FindFirstChild("Humanoid") then
