-- ============ ЛОГИКА ПОЛЁТА ============
uis.InputBegan:Connect(function(i, g)
    if g then return end
    local k = i.KeyCode
    if k == Enum.KeyCode.W then flyKeys.f = 1
    elseif k == Enum.KeyCode.S then flyKeys.b = 1
    elseif k == Enum.KeyCode.A then flyKeys.l = 1
    elseif k == Enum.KeyCode.D then flyKeys.r = 1
    end
end)

uis.InputEnded:Connect(function(i)
    local k = i.KeyCode
    if k == Enum.KeyCode.W then flyKeys.f = 0
    elseif k == Enum.KeyCode.S then flyKeys.b = 0
    elseif k == Enum.KeyCode.A then flyKeys.l = 0
    elseif k == Enum.KeyCode.D then flyKeys.r = 0
    end
end)

local function setFly(on)
    flyState = on
    local c = p.Character
    if not c then return end
    local torso = c:FindFirstChild("UpperTorso") or c:FindFirstChild("Torso")
    local h = c.Humanoid
    if not torso or not h then return end
    
    if flyState then
        vel = Instance.new("BodyVelocity", torso)
        vel.maxForce = Vector3.new(1e9, 1e9, 1e9)
        gyro = Instance.new("BodyGyro", torso)
        gyro.maxTorque = Vector3.new(1e9, 1e9, 1e9)
        gyro.P = 9e4
        h.PlatformStand = true
        flyBtn.Text = "🔴 ПОЛЁТ АКТИВЕН"
        flyBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 100)
    else
        if vel then vel:Destroy() end
        if gyro then gyro:Destroy() end
        h.PlatformStand = false
        flyBtn.Text = "🛸 АКТИВИРОВАТЬ ПОЛЁТ"
        flyBtn.BackgroundColor3 = Color3.fromRGB(200, 50, 150)
    end
end

rs.Heartbeat:Connect(function()
    if not flyState or not p.Character then return end
    local cam = workspace.CurrentCamera
    local move = Vector3.new(flyKeys.r - flyKeys.l, 0, flyKeys.f - flyKeys.b)
    if move.Magnitude > 0 then move = move.Unit end
    local velVec = (cam.CFrame.RightVector * move.X + cam.CFrame.LookVector * move.Z) * 50 * flySpeed
    if vel then vel.Velocity = velVec end
    if gyro then gyro.CFrame = cam.CFrame end
end)

local function switchTab(tabName)
    activeTab = tabName
    flyContainer.Visible = (tabName == "FLY")
    settingsContainer.Visible = (tabName == "SETTINGS")
    
    for name, btn in pairs(tabBtns) do
        if name == tabName then
            btn.BackgroundTransparency = 0
            btn.TextColor3 = Color3.fromRGB(255, 100, 200)
        else
            btn.BackgroundTransparency = 0.5
            btn.TextColor3 = Color3.fromRGB(200, 200, 200)
        end
    end
end

tabBtns["FLY"].MouseButton1Click:Connect(function() switchTab("FLY") end)
tabBtns["SETTINGS"].MouseButton1Click:Connect(function() switchTab("SETTINGS") end)

-- ПЕРЕТАСКИВАНИЕ
local drag = {active = false, start = nil, pos = nil}
bgGradient.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        drag.active = true
        drag.start = input.Position
        drag.pos = bgGradient.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                drag.active = false
            end
        end)
    end
end)

uis.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and drag.active then
        local delta = input.Position - drag.start
        bgGradient.Position = UDim2.new(drag.pos.X.Scale, drag.pos.X.Offset + delta.X, drag.pos.Y.Scale, drag.pos.Y.Offset + delta.Y)
        originalPos = bgGradient.Position
    end
end)
