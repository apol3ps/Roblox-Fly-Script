-- Логика полёта (та же, но с визуальными фишками)
local fly = false
local vel, gyro
local speed = 1
local keys = {f=0,b=0,l=0,r=0}

-- Эффект при включении полёта
local function flyFlash()
    local flash = Instance.new("Frame", f)
    flash.Size = UDim2.new(1, 0, 1, 0)
    flash.BackgroundColor3 = Color3.fromRGB(100, 0, 255)
    flash.BackgroundTransparency = 0.8
    flash.BorderSizePixel = 0
    ts:Create(flash, TweenInfo.new(0.5), {BackgroundTransparency = 1}):Play()
    game:GetService("Debris"):AddItem(flash, 0.5)
end

-- Управление
uis.InputBegan:Connect(function(i)
    local k = i.KeyCode
    if k == Enum.KeyCode.W then keys.f = 1
    elseif k == Enum.KeyCode.S then keys.b = 1
    elseif k == Enum.KeyCode.A then keys.l = 1
    elseif k == Enum.KeyCode.D then keys.r = 1
    end
end)
uis.InputEnded:Connect(function(i)
    local k = i.KeyCode
    if k == Enum.KeyCode.W then keys.f = 0
    elseif k == Enum.KeyCode.S then keys.b = 0
    elseif k == Enum.KeyCode.A then keys.l = 0
    elseif k == Enum.KeyCode.D then keys.r = 0
    end
end)

local function setFly(on)
    fly = on
    local c = p.Character
    if not c then return end
    local torso = c:FindFirstChild("UpperTorso") or c:FindFirstChild("Torso")
    local h = c.Humanoid
    if not torso or not h then return end
    
    if fly then
        flyFlash()
        vel = Instance.new("BodyVelocity", torso)
        vel.maxForce = Vector3.new(1e9, 1e9, 1e9)
        gyro = Instance.new("BodyGyro", torso)
        gyro.maxTorque = Vector3.new(1e9, 1e9, 1e9)
        gyro.P = 9e4
        h.PlatformStand = true
        h:SetStateEnabled(Enum.HumanoidStateType.Jumping, false)
        buttons.FLY.BackgroundColor3 = Color3.fromRGB(0, 200, 100)
    else
        if vel then vel:Destroy() end
        if gyro then gyro:Destroy() end
        h.PlatformStand = false
        h:SetStateEnabled(Enum.HumanoidStateType.Jumping, true)
        buttons.FLY.BackgroundColor3 = Color3.fromRGB(255, 50, 100)
    end
end

-- Цикл полёта
game:GetService("RunService").Heartbeat:Connect(function()
    if not fly or not p.Character then return end
    local cam = workspace.CurrentCamera
    local move = Vector3.new(keys.r - keys.l, 0, keys.f - keys.b)
    if move.Magnitude > 0 then move = move.Unit end
    local velVec = (cam.CFrame.RightVector * move.X + cam.CFrame.LookVector * move.Z) * 50 * speed
    if vel then vel.Velocity = velVec end
    if gyro then gyro.CFrame = cam.CFrame end
end)

-- Кнопки
buttons.FLY.MouseButton1Click:Connect(function() setFly(not fly) end)

buttons.UP.MouseButton1Down:Connect(function()
    local con = game:GetService("RunService").RenderStepped:Connect(function()
        if p.Character and p.Character.HumanoidRootPart then
            p.Character.HumanoidRootPart.CFrame = p.Character.HumanoidRootPart.CFrame * CFrame.new(0, 2, 0)
        end
    end)
    buttons.UP.MouseLeave:Connect(function() con:Disconnect() end)
end)

buttons.DOWN.MouseButton1Down:Connect(function()
    local con = game:GetService("RunService").RenderStepped:Connect(function()
        if p.Character and p.Character.HumanoidRootPart then
            p.Character.HumanoidRootPart.CFrame = p.Character.HumanoidRootPart.CFrame * CFrame.new(0, -2, 0)
        end
    end)
    buttons.DOWN.MouseLeave:Connect(function() con:Disconnect() end)
end)

buttons["+"].MouseButton1Click:Connect(function()
    speed = math.min(speed + 1, 20)
    speedLabel.Text = speed.."x"
    -- Визуальный фидбек
    speedLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
    task.wait(0.1)
    speedLabel.TextColor3 = Color3.fromRGB(0, 255, 200)
end)

buttons["-"].MouseButton1Click:Connect(function()
    speed = math.max(speed - 1, 1)
    speedLabel.Text = speed.."x"
    speedLabel.TextColor3 = Color3.fromRGB(255, 200, 100)
    task.wait(0.1)
    speedLabel.TextColor3 = Color3.fromRGB(0, 255, 200)
end)

-- Перерождение
p.CharacterAdded:Connect(function()
    task.wait(0.5)
    fly = false
    setFly(false)
end)

-- Перетаскивание окна
local dragging, dragInput, dragStart, startPos
f.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        dragging = true
        dragStart = input.Position
        startPos = f.Position
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)
f.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement and dragging then
        local delta = input.Position - dragStart
        f.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end
end)
